function calculate(){ let bore = document.getElementById("bore").value let rod = document.getElementById("rod").value let pressure = document.getElementById("pressure").value let boreArea = Math.PI*(bore/2)*(bore/2) let rodArea = Math.PI*(rod/2)*(rod/2) let extendForceN = boreArea*pressure*0.1 let retractForceN = (boreArea-rodArea)*pressure*0.1 // Convert Newtons to pounds-force let extendForceLbf = extendForceN * 0.224809 let retractForceLbf = retractForceN * 0.224809 // Convert to ft-lb equivalent torque at 1 ft let extendFtLb = extendForceLbf * 1 let retractFtLb = retractForceLbf * 1 document.getElementById("output").innerHTML = "Extend Force: "+extendForceN.toFixed(2)+" N
"+ extendForceLbf.toFixed(2)+" lbf
"+ extendFtLb.toFixed(2)+" ft-lb (1 ft lever)

"+ "Retract Force: "+retractForceN.toFixed(2)+" N
"+ retractForceLbf.toFixed(2)+" lbf
"+ retractFtLb.toFixed(2)+" ft-lb (1 ft lever)" }
